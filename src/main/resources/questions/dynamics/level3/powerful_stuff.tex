\begin{problem}[AO1989PIQ1l] 
{Which of the following is a unit of power?
\begin{enumerate}
	\item \quantity{}{N\,s\sup{-1}}
	\item \quantity{}{N\,m\,s}
	\item \quantity{}{N\,m\,s\sup{-1}}\answer
	\item \quantity{}{N\,m\,s\sup{-2}}
	\item \quantity{}{kg\,m\,s\sup{-1}}
\end{enumerate}
}
{\stress{Used with permission from UCLES, AO Level Physics, June 1989, Paper 1, Question 1.}}
{The correct answer is (c). This can be seen from \valuedef{P}{\vtr{F}\cdot\vtr{v}}{} or a similar formula which gives power; the units of \vari{F} are \quantity{}{N}, and the units of \vari{v} are \quantity{}{m\,s\sup{-1}}.}
\end{problem}
