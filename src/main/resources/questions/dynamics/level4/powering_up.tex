\begin{problem}[IntA1988PIQ5l]
\question{What is the power required to give a body of mass \vari{m} a forward acceleration \vari{a} when it is moving with velocity \vari{v} up a frictionless track inclined at an angle \vari{\theta} to the horizontal?}
\begin{enumerate}
	\item \choice[a]{\vari{mavg\sin(\theta)}}
	\item \choice[b]{\vari{mav\sin(\theta) + mgv}}
	\item \choice[c]{\vari{mav + mgv\sin(\theta)}}\correct
	\item \choice[d]{\vari{(mav + mgv)\sin(\theta)}}
	\item \choice[e]{\vari{\frac{mav + mgv}{\sin(\theta)}}}
\end{enumerate}
}
{\stress{Used with permission from UCLES, A Level Physics, November 1988, Paper 1, Question 5.}}
\answer{The correct answer is (c). This comes from considering forces up the plane: \valuedef{F - mg\sin(\theta)}{ma}{}, where \vari{F} is the forward force from the engine. This force is linked to power by \vair{P} $=$ \valuedef{\vtr{F} \cdot \vtr{v}}{Fv}{} and so \valuedef{F}{\frac{P}{v}}{}. Rearranging the initial expression will give the correct answer.}
\end{problem}
