\begin{problem}[IntA1987PIQ3l] 
{\question{Which of the following pairs contains one vector quantity and one scalar quantity?}
\begin{enumerate}
	\item \choice[a]{Displacement \quad : \quad Acceleration}
	\item \choice[b]{Force \quad : \quad Kinetic Energy}\correct
	\item \choice[c]{Momentum \quad : \quad Velocity}
	\item \choice[d]{Power \quad : \quad Speed}
	\item \choice[e]{Work \quad : \quad Potential Energy}
\end{enumerate}}
{\stress{Used with permission from UCLES, A Level Physics, November 1987, Paper 1, Question 3.}}
{\answer{The correct answer is (b). Displacement, acceleration, force, momentum and velocity are all vector quantities. Kinetic Energy, power, speed, work and potential energy are all scalar quantities.}}
\end{problem}
