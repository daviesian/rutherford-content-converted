\begin{problem}[A1988PIQ4l]
{\exposition{Two bodies, P and Q, of equal mass are travelling towards one another on a level frictionless track, with speeds \vari{u} and \vari{v} respectively. They make an elastic collision. At some instant during the collision, P is brought instantaneously to rest.} \question{What is the speed of Q at this instant?}}
\begin{enumerate}
	\item \choice[a]{zero}
	\item \choice[b]{\vari{(v - u)}}\correct
	\item \choice[c]{\vari{2(v - u)}}
	\item \choice[d]{\vari{\frac{1}{2}(v - u)}}
	\item \choice[e]{\vari{\sqrt{uv}}}
\end{enumerate}
}
{\stress{Used with permission from UCLES, A Level Physics, June 1988, Paper 1, Question 4.}}
\answer{The correct answer is (b). This can be obtained by using the conservation of momentum; the total momentum is \vari{m(v - u)} for all time, and if P has no momentum then Q has this total momentum and so the speed in (b).}
\end{problem}
