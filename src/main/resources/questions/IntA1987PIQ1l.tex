\begin{problem}[IntA1987PIQ1l] %{A given mass undergoes uniform acceleration when the resultant force acting on it:
{A toboggan is sliding down a hillside with a uniform acceleration. The resultant force acting on the toboggan:
\begin{enumerate}
	\item is zero.
	\item is constant.
	\item increases uniformly with time.
	\item decreases uniformly with time.
	\item is proportional to the displacement from a fixed point.
\end{enumerate}
}
%{\textit{Used with permission from UCLES, A Level Physics, November 1987, Paper 1, Question 1.}}
{\textit{Adapted with permission from UCLES, A Level Physics, November 1987, Paper 1, Question 1.}}
{The correct answer is (b). This just relies on a statement of Newton's Second Law.}
\end{problem}
