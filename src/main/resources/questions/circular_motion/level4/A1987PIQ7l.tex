\begin{problem}[A1987PIQ7l] 
{The second hand of a large clock is \quantity{3.0}{m} long. What is its mean angular speed?
\begin{enumerate}
	\item \quantity{1.4 \times 10^{-4}}{rad\,s$\sup{-1}$}
	\item \quantity{1.7 \times 10^{-3}}{rad\,s$\sup{-1}$}
	\item \quantity{5.2 \times 10^{-3}}{rad\,s$\sup{-1}$}
	\item \quantity{1.0 \times 10^{-1}}{rad\,s$\sup{-1}$} \answer
	\item \quantity{3.0 \times 10^{-1}}{rad\,s$\sup{-1}$}
\end{enumerate}}
{\stress{Used with permission from UCLES, A Level Physics, June 1987, Paper 1, Question 7.}}
{The correct answer is (d). This follows from the fact that there are \vari{2\pi} radians in a full circle, and the hand goes round every \quantity{60}{s}. Hence the mean angular speed is given by \value{\omega}{\frac{2\pi}{60}} = \quantity{0.10472}{rad\,s$\sup{-1}$} $\approx$ \quantity{1.0 \times 10\sup{-1}}{rad\,s$\sup{-1}$}. We can also see some of the information is irrelevant as the radius of the hand has no bearing on its angular speed.}
\end{problem}
