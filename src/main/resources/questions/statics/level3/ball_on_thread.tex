\begin{problem}[IntA1983PIIQ1a]
{\question{A small ball of mass \vari{m} is suspended by a light thread. When a strong wind blows horizontally, exerting a constant force F on the ball, the thread makes an angle \vari{\theta} to the vertical. Which one of the following expressions correctly defines \vari{\theta}?}
\begin{enumerate}
	\item \choice[a]{\valuedef{\cos{\theta}}{\frac{F}{mg}}{}}
	\item \choice[b]{\valuedef{\tan{\theta}}{\frac{F}{mg}}{}}\correct
	\item \choice[c]{\valuedef{\sin{\theta}}{\frac{F}{mg}}{}}
	\item \choice[d]{\valuedef{\cos{\theta}}{\frac{mg}{F}}{}}
	\item \choice[e]{\valuedef{\tan{\theta}}{\frac{mg}{F}}{}}
\end{enumerate}
}
{\textit{Used with permission from UCLES, A Level Physics, November 1983, Paper 2, Question 1.}}
{\answer{The correct answer is (b). The ball is kept in equilibrium by three forces: its weight acting downwards, the horizontal force exerted by the wind, and the tension exerted by the thread at an angle of \vari{\theta} to the vertical. Resolve horizontally to obtain \valuedef{F}{T\sin{\theta}}{}, and resolve vertically to obtain \valuedef{mg}{T\cos{\theta}}{}. Divide the first equation by the second to get the answer, \valuedef{\tan{\theta}}{\frac{F}{mg}}{}.}
}
\end{problem}
