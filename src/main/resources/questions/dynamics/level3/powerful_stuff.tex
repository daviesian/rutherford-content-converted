\begin{problem}[AO1989PIQ1l] 
{\question{Which of the following is a unit of power?}
\begin{enumerate}
	\item choice[a]{\quantity{}{N\,s\sup{-1}}}
	\item choice[b]{\quantity{}{N\,m\,s}}
	\item choice[c]{\quantity{}{N\,m\,s\sup{-1}}}\correct
	\item choice[d]{\quantity{}{N\,m\,s\sup{-2}}}
	\item choice[e]{\quantity{}{kg\,m\,s\sup{-1}}}
\end{enumerate}
}
{\stress{Used with permission from UCLES, AO Level Physics, June 1989, Paper 1, Question 1.}}
\answer{The correct answer is (c). This can be seen from \valuedef{P}{\vtr{F}\cdot\vtr{v}}{} or a similar formula which gives power; the units of \vari{F} are \quantity{}{N}, and the units of \vari{v} are \quantity{}{m\,s\sup{-1}}.}
\end{problem}
