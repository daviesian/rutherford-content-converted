%Physics: EnergyCons, KE, Power
\begin{problem}[IntA1986PIQ4] 
{A car of mass \vari{m} has an engine which can deliver power \vari{P}. What is the minimum time in which the car can be accelerated from rest to speed \vari{v}?
\begin{enumerate}
	\item \vari{\frac{mv}{P}}
	\item \vari{\frac{P}{mv}}
	\item \vari{\frac{mv^{2}}{2P}} \answer
	\item \vari{\frac{2P}{mv^{2}}}
	\item \vari{\frac{mv^{2}}{4P}}
\end{enumerate}}
{\stress{Used with permission from UCLES, A Level Physics, November 1986, Paper 1, Question 4.}}
{The correct answer is (c). This can be seen since Energy Gained = Power $\times$ Time, and the energy we want to gain is the kinetic energy at speed \vari{v}, equal to \vari{\frac{1}{2}mv^{2}}. This gives \valedef{Pt}{\frac{1}{2}mv^{2}}{}, which rearranges to the correct answer.}
\end{problem}
