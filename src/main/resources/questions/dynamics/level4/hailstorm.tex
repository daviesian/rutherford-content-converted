\begin{problem}[IntA1989PIQ24l]
{A surface is bombarded by particles, each of mass \vari{m}, which have velocity \vari{v} normal to the surface. On average, \vari{n} particles strike a unit area each second and rebound elastically. What is the pressure on the surface?
\begin{enumerate}
	\item \vari{nmv}
	\item \vari{2nmv} \answer
	\item \vari{\frac{1}{3}nmv^{2}}
	\item \vari{\frac{1}{2}nmv^{2}}
	\item \vari{nmv^{2}}
\end{enumerate}
}
{\stress{Used with permission from UCLES, A Level Physics, November 1989, Paper 1, Question 24.}}
{The correct answer is (b). This follows from \valuedef{F}{\frac{\d p}{\d t}}{}, with the momentum change of each particle being \valuedef{(mv) - (-mv)}{2mv}{} and since \vari{n} is particles per unit area per unit time it will convert the momentum change into a pressure: \vari{\Delta p \times \frac{\text{Number}}{\Delta t \times \text{Area}}} $=$  \valuedef{\frac{\text{Total Force}}{\text{Area}}}{\text{Pressure}}{}, where \valuedef{\text{Total Force}}{\frac{\Delta p}{\Delta t}}{}.
}
\end{problem}
