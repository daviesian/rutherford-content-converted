\begin{problem}[A1988PIQ1]  %Diagram removed
{\exposition{A picture weighing 10.0 N hangs freely by two strings attached to a hook above the picture. Each string forms an angle of 60$^{\circ}$ with the top of the picture frame.} \question{What is the tension in one of the strings?}
\begin{enumerate}
	\item \choice[a]{2.9 N}
	\item \choice[b]{5.0 N}
	\item \choice[c]{5.8 N}\correct
	\item \choice[d]{8.7 N}
	\item \choice[e]{10.0 N}
\end{enumerate}
}
{\textit{Used with permission from UCLES, A Level Physics, June 1988, Paper 1, Question 1.}}
{\answer{The correct answer is (c).} Resolve vertically on the picture to get $2T\sin{60^{\circ}} = W$, where W is the weight of the picture. Solve for $T = \frac{W}{2\sin{60^{\circ}}}$. $T = \frac{10}{2 \sin{60^{\circ}}} = 5.77$ N.
}
\end{problem}
