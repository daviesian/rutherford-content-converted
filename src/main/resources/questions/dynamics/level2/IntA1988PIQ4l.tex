\begin{problem}[IntA1988PIQ4l]
{Two masses, X and Y, are moving directly towards one another along a line. The momentum of X before the collision is \quantity{20}{kg\,m\,s$\sup{-1}$} and the momentum of Y before the collision is \quantity{12}{kg\,m\,s$\sup{-1}$}. After the collision, the masses move in opposite directions and the momentum of X is \quantity{2}{kg\,m\,s$\sup{-1}$}. What is the magnitude of the momentum of Y?
\begin{enumerate}
	\item \quantity{6}{kg\,m\,s$\sup{-1}$}
	\item \quantity{8}{kg\,m\,s$\sup{-1}$}
	\item \quantity{10}{kg\,m\,s$\sup{-1}$} \answer
	\item \quantity{30}{kg\,m\,s$\sup{-1}$}
	\item \quantity{34}{kg\,m\,s$\sup{-1}$}
\end{enumerate}
}
{\stress{Used with permission from UCLES, A Level Physics, November 1988, Paper 1, Question 4.}}
{The correct answer is (c). The total momentum in the direction X is moving in is \quantity{8}{kg\,m\,s$\sup{-1}} initially, so when X reverses direction and so has a momentum of \quantity{-2}{kg\,m\,s$\sup{-1}}; to make the total equal \quantity{8}{kg\,m\,s$\sup{-1}} and conserve momentum, the momentum of Y must be \quantity{10}{kg\,m\,s$\sup{-1}}.}
\end{problem}
