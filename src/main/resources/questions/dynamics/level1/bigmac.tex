\begin{problem}[Robin10]
\exposition{A Big Mac has \quantity{2.2}{MJ} of chemical potential energy which can be released by eating it.}
\begin{enumerate}
	\item \question[a]{How much energy is needed to lift a \quantity{50}{kg} student upwards through a height of \quantity{0.4}{m}?}
	\item \question[b]{How many step-ups of \quantity{0.4}{m} would the student need to make in order to burn off the energy of a Big Mac?}
	\item \question[c]{How many Big Macs would he need to consume in order to reach the top of Mount Everest, a height of \quantity{8}{km}?}
	\item \question[d]{If the energy of a Big Mac could be used by itself to propel itself upwards, how high could it rise? (The mass of a Big Mac is \quantity{0.22}{kg}.)}
\end{enumerate}
