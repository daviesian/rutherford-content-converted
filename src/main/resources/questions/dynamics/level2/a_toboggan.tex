\begin{problem}[IntA1987PIQ1l] %{A given mass undergoes uniform acceleration when the resultant force acting on it:
{\exposition{A toboggan is sliding down a hillside with a uniform acceleration.} \question{What is the resultant force acting on the toboggan?}
\begin{enumerate}
	\item \choice[a]{It is zero.}
	\item \choice[b]{It is constant.}\correct
	\item \choice[c]{It increases uniformly with time.}
	\item \choice[d]{It decreases uniformly with time.}
	\item \choice[e]{It is proportional to the displacement from a fixed point.}
\end{enumerate}
}
%{\textit{Used with permission from UCLES, A Level Physics, November 1987, Paper 1, Question 1.}}
{\stress{Adapted with permission from UCLES, A Level Physics, November 1987, Paper 1, Question 1.}}
{\answer{The correct answer is (b). This just relies on a statement of Newton's Second Law.}}
\end{problem}
