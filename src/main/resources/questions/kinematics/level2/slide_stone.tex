\begin{problem}[A1988PIQ2l] 
{\exposition{A stone slides across an icy surface and travels a distance \vari{x} in a time \vari{t}, while undergoing a constant deceleration.} \question{Which of the following pairs of quantities would give a straight line graph when plotted to represent the motion of the stone?}
\begin{enumerate}
	\item \choice[a]{\vari{x} and \vari{\frac{1}{t^{2}}}}
	\item \choice[b]{\vari{x} and \vari{\frac{1}{t}}}
	\item \choice[c]{\vari{\frac{x}{t}} and \vari{t}}\correct
	\item \choice[d]{\vari{\frac{x}{t^{2}}} and \vari{t}}
	\item \choice[e]{\vari{\frac{x}{t^{3}}} and \vari{t}}
\end{enumerate}
}
{\textit{Used with permission from UCLES, A Level Physics, June 1988, Paper 1, Question 2.}}
{\answer{The correct answer is (c).\nll
The SUVAT equation $s = ut + \frac{1}{2}at^{2}$ gives that the distance and time are related by $x = ut + \frac{1}{2}at^{2}$; but this has two terms dependent on \vari{t} on the right hand side. Dividing by \vari{t} gives the equation of a straight line $\frac{x}{t} = \frac{1}{2}at + u$.}
\end{problem}
